 BOARDCOMPUTER:
	
	MOV IN0,A;//000000P
	MOV A,B;//000000P

	MOV IN1,A;//000000C
	AND B,A; //PORTA AND CINTO
	MOV A,B;//B=PORTA AND CINTO

	MOV IN2,A;
	NOT A,A;//NEGA MARCHA
	MOV A,C;//C=NEGA MARCHA
	
	MOV IN3,A;//000000E
	OR C,A;//NEGA MARCHA OR EMBREAGEM	
	AND B,A;//PORTA AND CINTO AND NEGA MARCHA OR EMBREAGEM
	MOV A,OUT3;// LIGAR IGNICAO

	JMP BOARDCOMPUTER;
